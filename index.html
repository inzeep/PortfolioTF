Descripción: "Deathrun" fue un proyecto en conjunto a Bautista Brandan que consiste en un juego de Parkour plataformero 3D. (inspirado en el movimiento y mapping antiguo de los mapas custom de Counter Strike 1.6)
La mayor inspiración fueron los deathruns antiguos del CS 1.6, dónde un error podría costarte un gran progreso.
Eres un aventurero y tu objetivo es llegar al final de la edificación en ruinas y encontrar una reliquia. Este proyecto fue mi mayor trabajo relacionado a la creación de niveles y mi primera vez trabajando con un sistema de movimiento dínamico.

Codigo y explicación:

Muestra: Script de movimiento: Funciones de CharacterController como, movimiento, velocidad al caminar, velocidad al correr, altura del salto, gravedad, sensibilidad del mouse, cantidad de saltos, impulso del salto hacia delante, duración del impulso, la cámara y las particulas. Movimiento y rotación del mouse para todas las direcciones.
Movimiento del personaje y sprint alternables, gravedad, rotación de cámara, salto y impulso del salto. comprueba si el personaje está en el suelo y reinicia la cantidad de saltos posibles, permite movimiento, rotación de cámara y un ciclo for que llama un particle system si se produce un salto. 

personaje: using UnityEngine;

public class NewMonoBehaviourScript : MonoBehaviour
{
    public float velocidadCaminar = 5f;
    public float velocidadCorrer = 10f;
    public float alturaSalto = 1f;
    public float gravedad = 1f;
    public float sensibilidadMouse = 2f;
    public int cantidadMaxSaltos = 2;
    public float impulsoAdelante = 1500f;
    public float duracionImpulso = 0.2f;

    public Camera camaraJugador;

    public ParticleSystem[] efectosDeSalto;

    private CharacterController characterController;
    private bool estaEnElSuelo;
    private float rotacionX = 0f;
    private float rotacionY = 0f;
    private Vector3 velocidad;
    private Vector3 direccionMovimientoActual = Vector3.zero;
    private int saltosRestantes;

    private Vector3 impulsoActivo = Vector3.zero;
    private float tiempoImpulsoRestante = 0f;

    void Start()
    {
        characterController = GetComponent();
        Cursor.lockState = CursorLockMode.Locked;
        saltosRestantes = cantidadMaxSaltos;
    }

    void Update()
    {
        estaEnElSuelo = characterController.isGrounded;

        if (estaEnElSuelo && velocidad.y < 0)
        {
            saltosRestantes = cantidadMaxSaltos;

            
            for (int i = 0; i < efectosDeSalto.Length; i++)
            {
                efectosDeSalto[i].Stop();
            }
        }

        ManejarMovimiento();

        if (Input.GetButtonDown("Jump") && saltosRestantes > 0)
        {
            SaltarConImpulso();
        }

        RotarConMouse();
    }

    private void ManejarMovimiento()
    {
        float horizontal = Input.GetAxis("Horizontal");
        float vertical = Input.GetAxis("Vertical");

        float currentSpeed = Input.GetKey(KeyCode.LeftShift) ? velocidadCorrer : velocidadCaminar;

        Vector3 direccion = new Vector3(horizontal, 0f, vertical);
        direccion = Quaternion.Euler(0, rotacionX, 0) * direccion;
        direccion = direccion.normalized * currentSpeed;

        direccionMovimientoActual = new Vector3(direccion.x, 0f, direccion.z);

        if (!estaEnElSuelo)
        {
            velocidad.y += -gravedad * Time.deltaTime;
        }
        else if (velocidad.y < 0)
        {
            velocidad.y = -2f;
        }

        if (tiempoImpulsoRestante > 0)
        {
            direccion += impulsoActivo;
            tiempoImpulsoRestante -= Time.deltaTime;

            if (tiempoImpulsoRestante <= 0)
            {
                impulsoActivo = Vector3.zero;
            }
        }

        direccion.y = velocidad.y;

        characterController.Move(direccion * Time.deltaTime);
    }

    private void RotarConMouse()
    {
        float mouseX = Input.GetAxis("Mouse X") * sensibilidadMouse;
        float mouseY = Input.GetAxis("Mouse Y") * sensibilidadMouse;

        rotacionX += mouseX;
        rotacionY -= mouseY;
        rotacionY = Mathf.Clamp(rotacionY, -90f, 90f);

        transform.rotation = Quaternion.Euler(0f, rotacionX, 0f);
        if (camaraJugador != null)
        {
            camaraJugador.transform.localRotation = Quaternion.Euler(rotacionY, 0f, 0f);
        }
    }

    private void SaltarConImpulso()
    {
        float velocidadSalto = Mathf.Sqrt(2f * alturaSalto * gravedad);
        velocidad.y = velocidadSalto;

        impulsoActivo = transform.forward * impulsoAdelante;
        tiempoImpulsoRestante = duracionImpulso;

     
        int indiceEfecto = cantidadMaxSaltos - saltosRestantes;
        if (indiceEfecto >= 0 && indiceEfecto < efectosDeSalto.Length)
        {
            efectosDeSalto[indiceEfecto].Play();
        }

        saltosRestantes--;
    }
}
